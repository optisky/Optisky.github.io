%matlab
clear;
word = '文字'; 
len = length(word);

img = imread('lena.jpeg');
%img = imresize(img, [100, 100]); % 如果图片太大，适当减小
[r, c, d] = size(img);

figure;

for i = 1 : r
    for j = 1 : c
       pos = mod((i-1)*c+j, len); % 循环选中文字
       if pos == 0
           pos = len;
       end
       t =  word(pos);
       text(j, -i, t, 'color', [img(i,j,1), img(i,j,2), img(i,j,3)], 'fontsize', 1.5);
    end
end

xlim([1,c]);
ylim([-r,1]);
axis off;
print('lena.png', '-dpng', '-r1200');
